# for frontend

### on bash command

$ cd frontend

### to install package

$ yarn install

### to run the frontend

$ npm run start

### to build the frontend

$ npm run build

### to test the frontend

$ npm test

# for backend

### on bash command

$ cd backend

### to install package

$ yarn install

### to run the backend

$ npm run start

### unit tests

$ npm run test

### e2e tests

$ npm run test:e2e

# to see the whole project on localhost:3000 port

### copy the build folder and replace the build folder in bakend, of course you can use the current build folder in backend
